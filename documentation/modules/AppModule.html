		FlexLayoutModule,
		AngularFireModule.initializeApp(firebaseDEVConfig),
		AppRoutingModule,
		FeatureModulesAnimalsModule,
		FeatureModulesDashboardModule,
	],
	providers: [],
	bootstrap: [AppComponent],
})
export class AppModule {}
